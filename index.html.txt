        async function addUserManually() {

        function exportDailyList(date, dayList) {
            const v = venues.find(v => v.id === state.venueId);
            const dName = daysHebrew[new Date(date).getDay()];
            let text = `*${v.name.toUpperCase()} - יום ${dName} ${date}*\n--------------------------\n\n`;
            [{id:'table',l:'שולחנות'},{id:'bar',l:'בר'},{id:'free',l:'כניסות חינם'}].forEach(t => {
                const filtered = dayList.filter(r => r.type === t.id);
                if (filtered.length > 0) {
                    text += `*🏠 ${t.l}:*\n`;
                    filtered.forEach((r, i) => { text += `${i + 1}. *${r.name.trim()}* | אנשים: ${r.pax} | מינ': ₪${r.minOrder} | יחצן: ${r.promoterName}\n`; });
                    text += `\n`;
                }
            });
            window.open(`https://wa.me/?text=${encodeURIComponent(text)}`, '_blank');
        }

        function sendWhatsAppToClient(id) {
            const res = state.reservations.find(r => r.id === id);
            const text = `שלום ${res.name}, הזמנתך מאושרת ל ${res.pax} אנשים. מחכים לכם!`;
            window.open(`https://wa.me/${res.phone.replace(/\D/g,'').replace(/^0/, '972')}?text=${encodeURIComponent(text)}`, '_blank');
        }

        async function saveBooking(e) {
            e.preventDefault();
            const data = {
                venueId: state.venueId, name: state.formData.clientName, phone: state.formData.clientPhone,
                type: state.formData.type, pax: Number(state.formData.pax), eventDate: state.selectedDate,
                minOrder: Number(state.formData.manualPrice || (state.formData.pax * 300)), 
                notes: state.formData.notes || "", promoterId: state.phone, promoterName: state.userName, status: 'active',
                updatedAt: firebase.firestore.FieldValue.serverTimestamp()
            };
            if (state.editingId) await db.collection('reservations').doc(state.editingId).update(data);
            else { data.arrived = false; data.actualRevenue = 0; await db.collection('reservations').add(data); }
            updateState({ isAdding: false, editingId: null, formData: { type: 'table', pax: '', clientName: '', clientPhone: '', manualPrice: null, notes: '' } });
        }

        function renderSummary(list) {
            const summary = {}; list.forEach(r => { const p = r.promoterName || 'Admin'; if (!summary[p]) summary[p] = { count: 0, z: 0 }; summary[p].count++; summary[p].z += Number(r.actualRevenue || 0); });

            const tableRows = Object.entries(summary).map(([name, data]) => {
                const comm = (data.z * 0.1).toFixed(0); totalComm += Number(comm);
                return `<tr><td class="p-2 font-black text-[10px] text-right">${name}</td><td class="p-2 text-center text-[9px]">${data.count}</td><td class="p-2 font-black text-green-400 text-[10px] text-right">₪${data.z}</td><td class="p-2 font-black text-amber-500 text-[10px] text-right">₪${comm}</td></tr>`;
            }).join('');
            return `<div class="max-w-4xl mx-auto space-y-2"><div class="flex justify-between items-center"><button onclick="updateState({showSummary: false})" class="bg-red-700/50 text-white px-3 py-1.5 rounded text-[9px] font-black uppercase tracking-widest shadow-lg">חזור</button><h2 class="text-[10px] font-black text-amber-500 uppercase italic">עמלות (10%)</h2></div><div class="glass rounded-xl overflow-hidden border border-white/10 shadow-xl"><table class="w-full text-white text-right"><thead class="bg-white/5 text-[8px] uppercase font-black text-slate-400 text-right"><tr><th class="p-2">יחצן</th><th class="p-2 text-center">הזמנות</th><th class="p-2">Z סופי</th><th class="p-2 text-right">עמלה</th></tr></thead><tbody class="divide-y divide-white/5">${tableRows}</tbody><tfoot class="bg-amber-500/10 font-black"><tr><td colspan="3" class="p-2 text-right uppercase text-[9px]">סה"כ:</td><td class="p-2 text-amber-500 text-xs text-right font-black font-black">₪${totalComm}</td></tr></tfoot></table></div></div>`;

        auth.onAuthStateChanged(user => {
            if (!user) { auth.signInAnonymously(); return; }
            db.collection('reservations').onSnapshot(snap => { 
                const newRes = snap.docs.map(d => ({ id: d.id, ...d.data() }));
                if (!isFirstLoad && newRes.length > lastCount) triggerAlert();
                lastCount = newRes.length; isFirstLoad = false;
                updateState({ reservations: newRes }); 
            });
            db.collection('users').onSnapshot(snap => { updateState({ users: snap.docs.map(d => ({ id: d.id, ...d.data() })) }); });
        });
